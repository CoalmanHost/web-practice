    public static Shape CreateFromMesh(Mesh mesh)
    {
        Shape shape = new GameObject().AddComponent<Shape>();
        shape.transform.position = Vector3.zero;
        if (mesh == null)
        {
            throw new ArgumentNullException("Mesh was null!");
        }
        List<Polygon> polygons = new List<Polygon>();
        Dictionary<MathPlane, List<int>> planesVertices = new Dictionary<MathPlane, List<int>>();
        for (int i = 0; i < mesh.triangles.Length; i += 3)
        {
            List<int> triangle = new List<int>() { mesh.triangles[i], mesh.triangles[i + 1], mesh.triangles[i + 2] };
            Vector3 a = mesh.vertices[triangle[0]], b = mesh.vertices[triangle[1]], c = mesh.vertices[triangle[2]];
            Vector3 normal = Vector3.Cross(b - a, c - a).normalized;
            float d = -(float)Math.Round(Vector3.Dot(normal, a), precision);
            MathPlane plane = new MathPlane(normal, d);
            if (!planesVertices.ContainsKey(plane))
            {
                planesVertices.Add(plane, new List<int>());
            }
            foreach (var point in triangle)
            {
                planesVertices[plane].Add(point);
            }
        }

        foreach (var plane in planesVertices.Keys)
        {
            StringBuilder infoBuilder = new StringBuilder($"({plane.a})x +({plane.b})y + ({plane.c})z + ({plane.d}) = 0 -> {plane.a + plane.b + plane.c}={-plane.d} : ");
            foreach (var point in planesVertices[plane])
            {
                infoBuilder.Append($"{point}, ");
            }
            Debug.Log(infoBuilder.ToString());
        }

        foreach (var plane in planesVertices.Keys)
        {
            Mesh polygonMesh = new Mesh();
            Vector3[] vertices = new Vector3[planesVertices[plane].Max() - planesVertices[plane].Min() + 1];
            int[] triangles = new int[planesVertices[plane].Count];
            foreach (var pointIndex in planesVertices[plane]) ;
            int min = planesVertices[plane].Min();
            for (int i = 0; i < planesVertices[plane].Count; i++)
            {
                if (!vertices.Contains(mesh.vertices[planesVertices[plane][i]]))
                {
                    vertices[planesVertices[plane][i] - min] = mesh.vertices[planesVertices[plane][i]];
                }
                triangles[i] = planesVertices[plane][i] - min;
            }
            polygonMesh.vertices = vertices;
            polygonMesh.triangles = triangles;
            GameObject polygonObject = GameObject.CreatePrimitive(PrimitiveType.Quad);
            polygonObject.transform.position = Vector3.zero;
            polygonObject.GetComponent<MeshFilter>().mesh = polygonMesh;
            Polygon polygon = polygonObject.AddComponent<Polygon>();
            polygon.parent = shape;
            polygon.skin = new Skin();
            polygon.skin.material = polygonObject.GetComponent<MeshRenderer>().material;
            polygons.Add(polygon);
        }
        foreach (var polygon in polygons)
        {
            polygon.transform.SetParent(shape.transform);
        }
        shape.polygons = polygons;
        shape.transform.localScale = Vector3.one * 0.2f;
        return shape;
    }
